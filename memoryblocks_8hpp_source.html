31 /** ************************************************************************************************
32  * Simple class that allocates a series of bigger memory blocks and provides sequential
33  * allocation of sub-ranges of those. Furthermore, the allocations can be reset and the
34  * allocated blocks reused.
35  *
36  * The size of the blocks allocated is defined with constructor parameter \p{pSize}.
37  * If an invocation of an allocation method is requesting a bigger allocation, then a new block
38  * with that given size is created.
39  *
40  * \note
41  * This class is new with \alib. It is not considered finished, optimized and stable in design.
42  **************************************************************************************************/
43 class MemoryBlocks
44 {
45  // #############################################################################################
46  // inner types
47  // #############################################################################################
48  protected:
49  /** ****************************************************************************************
50  * Internal type defining an allocated memory chunk.
51  ******************************************************************************************/
52  struct Block
53  {
54  char* mem; ///< The allocated memory. This is in fact a <c>void*</c> but
55  ///< <c>char*</c> allows arithmetics.
56  size_t size; ///< The size of the block.
57  size_t left; ///< The current fill.
58 
59  /** ************************************************************************************
60  * Constructor allocating the given \p{pSize} of memory.
61  * This memory is not freed with a destructor, therefore
62  * objects of this type need to be deleted manually.
63  *
64  * @param pSize The size of memory chunk to allocate
65  **************************************************************************************/
66  inline Block( size_t pSize )
67  : size(pSize)
68  , left(pSize)
69  {
70  mem= new char[pSize];
71  }
72 
73  /** ************************************************************************************
74  * Returns \c nullptr if the requested size is not available in the block.
75  * Otherwise, the pointer to the aligned free chunk of memory inside the block is
76  * returned.
77  *
78  * @param requestedSize The size to allocate.
79  * @param alignment The alignment.
80  * @return \c nullptr on failure, otherwise pointer to the requested memory.
81  **************************************************************************************/
82  inline char* get( size_t requestedSize, size_t alignment )
83  {
84  void* actMem= mem + size - left;
85  if( !std::align( alignment, requestedSize, actMem, left) )
86  return nullptr;
87  left-= requestedSize;
88 
89  return reinterpret_cast<char*>( actMem );
90  }
91  };

153  public:
154  /** ****************************************************************************************
155  * Allocates aligned memory of size of type \p{T}.
156  *
157  * @return A pointer to the allocated memory.
158  ******************************************************************************************/
159  template<typename T>
160  inline T* Alloc()
161  {
162  return reinterpret_cast<T*>( getAlignedChunk( sizeof(T[1]), alignof(T) ) );
163  }
164 
165  /** ****************************************************************************************
166  * Allocates aligned memory for an array of objects \p{T} of size \p{length}.
167  *
168  * @param length The number of objects in of type \p{T}.
169  * @tparam T The array element type.
170  * @tparam TSize The type of the array length value. This template parameter is provided
171  * for convenience, to avoid casts from singed types.
172  * @return A pointer to the allocated memory.
173  ******************************************************************************************/
174  template<typename T, typename TSize>
175  inline T* AllocArray( TSize length )
176  {
177  return reinterpret_cast<T*>( getAlignedChunk( sizeof(T[1]) * static_cast<size_t>(length), alignof(T[]) ) );
178  }

183  * In this case, future invocations of allocation methods will allocate new blocks from
