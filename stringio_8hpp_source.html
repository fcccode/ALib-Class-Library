30 /** ************************************************************************************************
31  * This class is a simple helper class that converts strings of the
32  * \ref alib_strings_templated "default string type" defined with the \alib compilation
33  * flags to narrow strings as expected by \c std::ostream objects.
34  *
35  * The basic goal of this class is to avoid preprocessor directives for code selection when
36  * the default string type of \alib is a wide string.
37  *
38  * For example, to write string data into a file, the following approach is advised:
39  *
40  * 1. Create an output stream of (derived) type \c std::ostream.
41  * 2. Create an object of this class and pass the output stream via #SetStream.
42  * 3. Write string data stored in objects of platform dependent type alias \ref aworx::AString
43  * by passing those to method #Write.
44  *
45  * Within step 3, the compiler chooses the right overloaded version of method #Write. Hence the
46  * potentially needed conversion of the string data is done transparently.
47  *
48  * \see Class #StringReader.
49  **************************************************************************************************/
50 class StringWriter
51 {
52  protected:
53  /// The string buffer used for conversion.
54  NAString converter;
55 
56  public:
57  /// The output stream as provided with #SetStream. Will be set to the \c std::cout,
58  /// respectively \c std::wcout in the constructor.
59  std::ostream* ostream;
60 
61  public:
62  /** ****************************************************************************************
63  * Constructor.
64  * Invokes #SetStream passing \c std::cout.
65  ******************************************************************************************/
66  inline StringWriter()
67  : ostream( &std::cout )
68  {}
69 
70 
71  /** ****************************************************************************************
72  * Sets the output stream.
73  *
74  * @param os Pointer to the output stream to write to.
75  ******************************************************************************************/
76  inline void SetStream( std::ostream * os) { ostream= os; }
77 
78  /** ****************************************************************************************
79  * Returns the output stream previously set with #SetStream.
80  *
81  * @return The output stream set with #SetStream.
82  ******************************************************************************************/
83  inline std::ostream* GetStream() { return ostream; }
84 
85  /** ****************************************************************************************
86  * Write the given narrow string to the stream.
87  * \note
88  * The return value is the length of the given string if it was converted to wide string,
89  * even in the case that it is not needed to be converted.
90  * This is useful to determine the real output width. Of-course, in many locales, this is
91  * still not the real output width, because even uni-code characters are not guaranteed
92  * to represent exactly one printable character. But still, this value is already a
93  * much better approximation than the length of the given narrow string.
94  *
95  * @param src The string to write.
96  * @return Returns the length of the given string as wide string.
97  ******************************************************************************************/
98  inline integer Write( const NString& src )
99  {
100  ostream->write( src.Buffer(), src.Length() );
101  return src.WStringLength();
102  }
103 
104  /** ****************************************************************************************
105  * Write the given wide string to the stream.
106  * @param src The string to write.
107  * @return Returns the length of the given wide string.
108  ******************************************************************************************/
109  inline integer Write( const WString& src )
110  {
111  converter.Clear() << src;
112  ostream->write( converter.Buffer(), converter.Length() );
113  return src.Length();
114  }
115 };
