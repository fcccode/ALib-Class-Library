130  /** ********************************************************************************************
131  * Translates exceptions thrown by the \alibmod_nolink_cli library to exit codes defined with
132  * \alib{cli,CLIApp::DefineExitCodes}.
133  *
134  * If the code is not found, this indicates an error in the resource data, as an exit
135  * code corresponding to the \alibmod_nolink_cli exceptions is obviously missing.
136  * In this case, \c -1 is returned. In debug compilations an \alib assertion is raised.
137  *
138  * @param cliApp The friend object we work on.
139  * @param exception The cli exception caught.
140  * @return The exit code to return to the caller. \c -1, if not found.
141  **********************************************************************************************/
142  static ALIB_API
143  int GetExitCode( CLIApp& cliApp, Exception& exception )
144  {
145  auto element= exception.Code().Get<lib::cli::Exceptions>();
146  for( auto& exitCode : cliApp.ExitCodeDecls )
147  if( exitCode.second.AssociatedCLIException() == element )
148  return exitCode.second.EnumBox.Value();
149  ALIB_ERROR( "No exit code associated with {}.", element );
150  return -1;
151  }
