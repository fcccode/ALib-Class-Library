87 /** ************************************************************************************************
88  * Template (meta programming) struct to determine if an arbitrary string type is terminatable or
89  * not.
90  * The 'default implementation' inherits from std::false_type
91  *
92  * For all user defined string types (from an \alib perspective external types), which
93  * - implement template struct
94  * \ref aworx::lib::strings::T_String "T_String"
95  * to be able to serve as a string argument for implicit construction of class
96  * \alib{strings,StringBase,String}
97  * - and which do not provide a terminated buffer in that <em>ToString</em> method
98  * - and which should in addition be suitable to serve as string argument to implicit constructor
99  * of class \alib{strings,TStringBase,TString}
100  *
101  * a specialization of this struct (that simply is derived from <em>std::true_type</em>)
102  * has to be provided.
103  *
104  * When this is done, the capacity for the termination character '\0' has to be available
105  * in the buffer that is passed in method
106  * \ref aworx::lib::strings::T_String::Buffer "T_String::Buffer".
107  * In other words, their buffer needs to be writable and at least one character longer than
108  * their content length.
109  *
110  * \note
111  * If (user defined) objects are terminated already when passed to the constructor of
112  * TString, then this template is <b>not</b> needed to be defined.<br>
113  * For example, the implementation of method <em>ToString</em> for type <em>std::string</em>
114  * returns <em>std::string::c_str()</em>. This way, the string returned is well terminated and
115  * there is no need to partially specialize this struct for type <em>std::string</em>.
116  **************************************************************************************************/
117 template<typename Type> struct T_IsTerminatable : public std::false_type { };

120 /** ************************************************************************************************
121  * This class specializes class
122  * \alib{strings,StringBase,String} to represent zero terminated or more precisely
123  * zero \e terminatable strings.
124  * Zero terminated strings are especially needed when string data has to be passed to system
125  * functions. Also, some efficient algorithms (platform dependent and mostly written in assembly
126  * language) exist. Hence, various functions and methods of \alib require a const reference
127  * to an object of this type \b %TString instead to an object of class \b %String.
128  *
129  * <b>Implicit construction</b><p>
130  * This class hides its parents' constructors and re-implements the template meta
131  * programming based "all-for-one" constructor
132  * \alib{strings::StringBase::StringBase(const TStringLike&),StringBase(const TStringLike&)} by introducing
133  * \alib{strings::TStringBase::TStringBase(const TTerminatable&),TString(const TTerminatable&)}.
134  *
135  * As it is not possible to implement compile time assertions for the save provision of
136  * terminatable types (without taking serious restrictions into account), this constructor
137  * performs a run time check (in debug compilations of \alib).<br>
138  * It is checked whether
139  * - the string type passed is already terminated or
140  * - can be terminated because a) the buffer reserved capacity for the termination value and b)
141  * the buffer is writable.
142  *
143  * The second check is done only if the first did not apply and is performed using partially
144  * implemented template struct
145  * \ref aworx::lib::strings::T_IsTerminatable "T_IsTerminatable". Consult this struct for information
146  * about how to enable external (user defined) string types to be used as parameters to implicitly
147  * construct objects of this class.
148  *
149  * \note
150  * While the good news is that C++ string literals, any zero terminated <em>char*</em> and
151  * objects of type <em>std::strings</em> or \alib's \b %AString,
152  * are well suited to implicitly construct respectively directly serve as objects of this class,
153  * objects of type
154  * \alib{strings,StringBase,String} itself and especially of type
155  * \alib{strings,SubstringBase,Substring} are not.
156  * Therefore the latter are accepted only if they are terminated already.
157  * But obviously, this is especially unlikely for objects of type
158  * \alib{strings,SubstringBase,Substring}.
159  *
160  * \note
161  * Therefore, for example to pass a <em>Substring</em>
162  * as a parameter to functions accepting only terminatable strings, it is possible to
163  * wrap them into temporary objects of type
164  * \alib{strings,AStringBase,AString} or, if maximum size is known at compile time,
165  * of type
166  * \alib{strings,PreallocatedStringBase,PreallocatedString}.
167  * Of-course, a performance drawback (for creating the temporary copy of the contents) has to
168  * be taken into account.
169  *
170  * \note
171  * Alternatively, class \b %Substring and class \b %String provide some useful
172  * implementations of slightly slower versions of some of the methods introduced this class.
173  * It has to be considered case by case which option is the best in respect to performance and
174  * memory use.
175  *
176  * <b>Constant Nature</b><p>
177  * Same as parent class
178  * \alib{strings,StringBase,String}, this class is not copying the data of the provided
179  * source. It is designed preliminary to offer a fast but convenient (by implicit conversion)
180  * type for function parameters that require terminated cstring buffers. As with parent class \b %String,
181  * the life-time of objects of this class is considerably short.<br>
182  * Only derived class \alib{strings,AStringBase,AString} and its different
183  * further specializations will copy the buffer and always include space for the termination
184  * character in their buffers' capacity.
185  *
186  * <b>Parsing Numbers</b><p>
187  * This class provides methods for parsing integer and floating point data.
188  * For implementation performance reasons, those are <em>not</em> provided by parent class \b %String
189  * which does not guarantee zero terminated strings. If parsing has to be done on non-zero
190  * terminated string data, consider using a slightly slower variant of the parsing methods which
191  * are found in class \alib{strings,SubstringBase,Substring}.<br>
192  * It is rather more efficient, to instantiate a local Substring object that just wraps an
193  * unterminated string of type \b %String and use its parsing interface, than copying the
194  * unterminated data into a terminatable string variant (e.g. \b %AString) and to use the
195  * interface provided herein.
196  *
197  * @tparam TChar The character type. Implementations for \c char and \c wchar_t are provided
198  * with type definitions \ref aworx::TString and \ref aworx::TString.
199  **************************************************************************************************/

359  /** ****************************************************************************************
360  * Returns the index of the first character which is included, respectively <em>not</em>
361  * included in a given set of characters.
362  *
363  * This method searches forwards. For backwards search, see
364  * \alib{strings,StringBase::LastIndexOfAny,String::LastIndexOfAny}.
365  *
366  * \note This method reimplements method
367  * \alib{strings,StringBase::IndexOf,String::IndexOf} of parent class.
368  * This implementation however needs a zero-terminatable string for the needles
369  * (beside the fact that this is a zero terminatable string itself).
370  * If no zero-terminatable needles are available, parent method accepting non-zero
371  * terminated needles needs to be invoked. This has to be done by explicitly naming
372  * the parent class in the invocation, e.g. <c>mystring.String::IndexOfAny(...)</c>.<br>
373  * On most platforms, this zero-terminated version should perform faster than
374  * the original method in class \b %String.
375  *
376  * @param needles Pointer to a zero terminated set of characters to be taken into
377  * account.
378  * @param startIdx The index to start the search at. If the given value is less than 0,
379  * it is set to 0. If it exceeds the length of the string, the length of
380  * the string is returned.
381  * Defaults to 0.
382  * @tparam TCheck Defaults to \c true which is the normal invocation mode.
383  * If \c <false> is added to the method name, no parameter checks are
384  * performed and the needles must not be empty.
385  * @tparam inclusion Denotes whether the search returns the first index that holds a value
386  * that is included or that is not excluded in the set of needle
387  * characters.
388  *
389  * @return The index of the first character found which is included, respectively not
390  * included, in the given set of characters.
391  * If nothing is found, -1 is returned.
392  ******************************************************************************************/
393  template <lang::Inclusion inclusion,
394  bool TCheck= true >
395  inline
396  integer IndexOfAny( const TStringBase& needles, integer startIdx= 0 )
397  const
398  {
399  if (TCheck)
400  {
401  if ( startIdx < 0 ) startIdx= 0;
402  if ( startIdx >= StringBase<TChar>::length ) return -1;
403  }
404  else
405  {
406  ALIB_ASSERT_ERROR( startIdx >= 0
407  && startIdx < StringBase<TChar>::length
408  && needles.Length() != 0,
409  ASTR("Non checking and illegal parameters") );
410  }
411 
412 
413  if ( inclusion == lang::Inclusion::Include )
414  {
415  integer idx= CString<TChar>::IndexOfAnyIncludedZT( ToCString() + startIdx, needles.ToCString() );
416  return idx >= 0 ? idx + startIdx : -1;
417  }
418  else
419  {
420  const TChar* haystack= ToCString() + startIdx;
421  integer idx= CString<TChar>::IndexOfAnyExcludedZT( haystack, needles.ToCString() ) ;
422  return *( haystack + idx ) == '\0' ? -1 : startIdx + idx;
423  }
424  }

445  /** ****************************************************************************************
446  * This instances' buffer will be terminated (if not already) and returned.
447  * If this object is \e nulled, a pointer to an empty cstring is returned.
448  * This allows to omit explicit checks for \e nulled objects, in the frequent case that
449  * no difference should be made in the processing of a \e nulled or a not \e nulled, but
450  * empty object.
451  *
452  * \note
453  * If used within loops, it is advisable to use a combination of #Terminate (before
454  * the loop) and #Buffer (inside the loop) to avoid the overhead for a repeated
455  * termination check.
456  *

463  return std::is_same<TChar,char>::value ? reinterpret_cast<const TChar*>( "" )
464  : reinterpret_cast<const TChar*>( L"" );

466  return StringBase<TChar>::buffer;
