31 /** ************************************************************************************************
32  * This class encapsulates a system dependent timer value of type \alib{time,Ticks} and provides
33  * some simple interface for measuring multiple time spans and providing their sum and average.
34  **************************************************************************************************/
35 class StopWatch
36 {
37  protected:
38 
39  /// The current start time.
40  Ticks startTime;
41 
42  /// The number of samples performed.
43  int cntSamples =0;
44

48

51  * Creates a started StopWatch.
52  ******************************************************************************************/
53  inline

56  , sum()
57  {}
58 
59  public:
60  /** ****************************************************************************************
61  * Provides access to the internal start time.
62  * @return The start time
63  ******************************************************************************************/
64  inline Ticks& GetStartTime()
65  {
66  return startTime;
67  }
68 
69  /** ****************************************************************************************
70  * Sets the start time to now.
71  * This affects both, the reference value for the calculation of this StopWatch's age in
72  * subsequent calls, as well as subsequent sample time spans.
73  ******************************************************************************************/
74  inline void Start()
75  {
76  startTime= Ticks::Now();
77  }
78 
79  /** ****************************************************************************************
80  * Sets the internal value to current system time and clears existing sum and quantity of
81  * samples.
82  ******************************************************************************************/
83  inline void Reset()
84  {
85  sum= Ticks::Duration::FromNanoseconds(0);
86  cntSamples= 0;
87  Start();
88  }
89 
90  /** ****************************************************************************************
91  * Returns the time span between the current system time and the internal start value.
92  * In addition this value is added to the sum of sample times and the sample counter is
93  * increased by one. Lastly the internal reference value is set to now. Therefore, a
94  * subsequent call to this function would measure the time span from this call to this
95  * subsequent call (if the internal start time value was not set differently meanwhile).
96  *
97  * @return The time difference between the current system time and the internal
98  * reference value.
99  ******************************************************************************************/
100  inline Ticks::Duration Sample()
101  {
102  Ticks::Duration sample= startTime.Age();
103  sum+= sample;
104  cntSamples++;
105  startTime= Ticks::Now();
106 
107  return sample;
108  }
109 
110  /** ****************************************************************************************
111  * Returns the number of calls to #Sample since this instance was created or #Reset was
112  * invoked.
113  * @return The number of samples.
114  ******************************************************************************************/
115  inline int GetSampleCnt() const { return cntSamples; }
116 
117  /** ****************************************************************************************
118  * Returns the cumulated time of all samples taken since this instance was created or
119  * cleared.
120  *
121  * @return The cumulated measured time.
122  ******************************************************************************************/
123  inline Ticks::Duration GetCumulated() const { return sum; }
124 
125  /** ****************************************************************************************
126  * Returns the average time of all samples since this instance was created or reset.
127  * If no measurement was performed, the result value will be set to \c 0.
128  *
129  * @return The cumulated measured time.
130  ******************************************************************************************/
131  inline Ticks::Duration GetAverage() const
132  {
133  return cntSamples== 0 ? Ticks::Duration()
134  : ( sum / static_cast<int64_t>(cntSamples) );
135  }
