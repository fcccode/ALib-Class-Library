17 /** ************************************************************************************************
18  * Simple string class used by native C++ \alib{expressions,CallbackDecl,expression callback functions}
19  * to allocate string data returned as (intermediate) expression result.
20  *
21  * Optional constructors allow to copy a source string or alternatively (any) boxed value.
22  * (Boxes provided need to have implemented boxing interface \alib{strings::boxing,IApply}.).
23  * The additional size needed besides optional copy data has to be known at the creation of the
24  * string. A \b %ScopeString must not exceed its capacity after creation. Doing so, results in a
25  * memory leak, as the string will not be properly destructed.
26  * In debug compilations an \alib warning is raised when capacity gets exceeded.
27  *
28  * \note
29  * If dynamic strings are needed, a custom \b %Scope class may be created that allows allocation
30  * of \e 'true' \b %AString objects, e.g. stored in a vector of pointers which are deleted
31  * upon clearing of the scope.
32  **************************************************************************************************/
33 class ScopeString : public strings::AStringBase<character>
34 {
35  public:
36 
37  /**
38  * Allocates a scope string of given capacity.
39  * @param scope The scope to allocated data with.
40  * @param capacity The capacity of the of the string.
41  */
42  ScopeString( Scope& scope, integer capacity )
43  : strings::AStringBase<character>( scope.Memory.AllocArray<character>( static_cast<size_t>(capacity + 1) ) ,
44  capacity + 1 )
45  {}
46 
47  /**
48  * Allocates a scope string of given capacity plus the length of the string \p{src} and copies
49  * \p{source} to it.
50  * @param scope The scope to allocated data with.
51  * @param src The source string to copy.
52  * @param additionalCapacity The extra capacity to allocate
53  */
54  ScopeString( Scope& scope, const strings::StringBase<character>& src, integer additionalCapacity )
55  : strings::AStringBase<character>( scope.Memory.AllocArray<character>( src.Length() + additionalCapacity + 1),
56  src.Length() + additionalCapacity + 1 )
57  {
58  strings::StringBase<character>::length= src.CopyTo( strings::StringBase<character>::vbuffer );
59  }
60 
61  /**
62  * Returns a string with contents of the given \p{box} written at the start (using boxing
63  * interface \alib{strings::boxing,IApply}) and the given additional capacity.
64  * @param scope The scope to allocated data with.
65  * @param additionalCapacity Additional capacity to allocate in \p{scope}.
66  * @param value A boxed object to write to the start of the string.
67  */
68  ScopeString( Scope& scope, integer additionalCapacity, const Box& value)
69  : strings::AStringBase<character>()
70  {
71  strings::PreallocatedStringBase<character,128> tmp;
72  ALIB_WARN_ONCE_PER_INSTANCE_DISABLE( tmp, ReplaceExternalBuffer )
73  tmp << value;
74 
75  strings::AStringBase<character>::SetBuffer( scope.Memory.AllocArray<character>( tmp.Length() + additionalCapacity + 1 ),
76  tmp.Length() + additionalCapacity + 1 ,
77  tmp.Length(), lang::Responsibility::KeepWithSender );
78 
79  tmp.CopyTo( strings::StringBase<character>::vbuffer );
80  }
81 };
