143 /** ************************************************************************************************
144  * This is a type applicable to objects of type
145  * \alib{strings,AStringBase,AString}. Various constructors consume integer and
146  * floating point types, along with formatting options. The specialization of struct
147  * \alib{strings,T_Apply}
148  * will use a given (or defaulted) instance of class
149  * \alib{strings,NumberFormatBase,NumberFormat} to append the value to the
150  * \b %AString in question.
151  *
152  * \note
153  * Within the same header file that this class in declared in, there are several
154  * specializations of \alib{strings,T_Apply} defined for plain integer and
155  * floating point types. These specializations create an object of this type, providing the
156  * value only, hence, using this classes constructors' default values. The number format
157  * used as default by the constructors of this class is
158  * \alib{strings,NumberFormatBase::Computational,NumberFormat::Computational}.
159  * As a result, the application of such core types, as in:
160  * \snippet "DOX_ALIB_APPLYTO.cpp" DOX_ALIB_APPLYTO_FORMAT1
161  * which produces:
162  * \snippet "DOX_ALIB_APPLYTO_FORMAT1.txt" OUTPUT
163  *
164  * \note
165  * does \b not use a locale specific number format. Instead it uses one that is exchangeable
166  * between applications independent from the locale setting of the executing machine.<br>
167  * Consequently, for locale specific output, an object of this class needs to be applied
168  * along with a locale enabled instance of \b %NumberFormat. For example:
169  * \snippet "DOX_ALIB_APPLYTO.cpp" DOX_ALIB_APPLYTO_FORMAT2
170  * which - dependent on the current local setting - might produce:
171  * \snippet "DOX_ALIB_APPLYTO_FORMAT2.txt" OUTPUT
172  *
173  *
174  * <b>Inner Types:</b><br>
175  * Besides accepting plain number types, this class in addition aggregates several public inner
176  * types, namely
177  * - \alib{strings,FormatBase::Tab,Format::Tab},
178  * - \alib{strings,FormatBase::Field,Format::Field},
179  * - \alib{strings,FormatBase::Escape,Format::Escape},
180  * - \alib{strings,FormatBase::Bin,Format::Bin},
181  * - \alib{strings,FormatBase::Hex,Format::Hex} and
182  * - \alib{strings,FormatBase::Oct,Format::Oct}.
183  *
184  * Each of these classes has partial implementations of template function
185  * \alib{strings,T_Apply} and provide certain formatting options.
186  *
187  * \note The types are not 'physically' related to this class. Instead, they have been
188  * aggregated here for 'esthetical' reasons.
189  *
190  * <b>Details on Formats:</b><br>
191  * Details on the options of formatting integer and floating point numbers are documented
192  * with class
193  * \alib{strings,NumberFormatBase,NumberFormat}.
194  *
195  * @tparam TChar The character type. Implementations for \c char and \c wchar_t are provided
196  * with type definitions \ref aworx::NFormat and \ref aworx::WFormat.
197  **************************************************************************************************/
198 template<typename TChar>

207  /** ********************************************************************************************
208  * Used to create temporary objects which are \alib{strings,T_Apply,applied} to
209  * \alib{strings,AStringBase,AString} instances.<br>
210  *
211  * Appends \e tab characters (as provided) to reach a certain length (aka tabulator position)
212  * of the \p{target}. The tab position is referenced to an optionally given \p{reference} value
213  * which might be the start of the string or the position of the last newline. If this
214  * parameter is negative, the last newline characters are searched from the end of the
215  * string backwards.<br>
216  * Referring to that as position 0, the tab position is then located at the next multiple
217  * of parameter \p{tabSize}, after having added \p{minPad} tab characters.
218  **********************************************************************************************/
219  struct Tab
220  {
221  public:
222  /// The tab positions are multiples of this value.
223  integer tabSize;
224 
225  /** The reference length of the AString which is taken as relative tab position
226  * (instead of the beginning of the string). If -1, the target %AString is
227  * searched backwards for the last newline and this position is used as the
228  * reference.
229  */
230  integer reference;
231 
232  /// The minimum pad characters to add. Defaults to 1.
233  integer minPad;
234 
235  /** The character to insert to reach the tab position. Normally this is the space
236  * character ' '.*/
237  TChar tabChar;
238 
239  /**
240  * Constructor. Copies the parameters.
241  *
242  * @param size The tab positions are multiples of this parameter.
243  * @param referenceIdx The reference index marking the start of the actual line.
244  * If -1, the last new-line character is searched from the end of
245  * the string backwards, and used. Defaults to 0.
246  * @param minPadChars The minimum pad characters to add. Defaults to 1.
247  * @param fillChar The character to insert to reach the tab position.
248  * Defaults to ' ' (space).
249  */
250  Tab( integer size, integer referenceIdx = 0, integer minPadChars= 1, TChar fillChar= ' ' )
251  : tabSize(size), reference(referenceIdx), minPad(minPadChars), tabChar(fillChar)
252  {}
253  };

257  * \alib{strings,AStringBase,AString} instances.<br>

259  * Appends the given object to the AString using a defined 'field'-width.
260  * If the contents of the field is shorter than parameter \p{width} specifies, the field is
261  * filled with a corresponding amount of \p{padChar} characters.<br>
262  * Parameter \p{alignment} of type
263  * \ref aworx::lib::lang::Alignment "Alignment" allows to left-, right- or center-align
264  * the contents of the field.

267  * parameter will be of type <c>const String&</c>.<br>

269  * the given box internally to receive the string representation.

272  * to be formatted in a field, this boxing interface has to be implemented. As

274  * objects already, all that is needed is to use macro

276  * section of an application.
