57 /**
58  * Struct used as parent for types
59  * - \alib{cli,CommandDecl},
60  * - \alib{cli,OptionDecl},
61  * - \alib{cli,ParameterDecl} and
62  * - \alib{cli,ExitCodeDecl}.
63  *
64  * Stores runtime information of user-defined enum types.
65  *
66  * Construction is done by passing a custom enum element of an enum type equipped with
67  * \alib{lang,T_EnumMetaDataDecl,enum meta data} of a distinct type.
68  *
69  * Such enums are announced to <b>%ALib CLI</b> using macros
70  * - \ref ALIB_CLI_COMMANDS,
71  * - \ref ALIB_CLI_OPTIONS,
72  * - \ref ALIB_CLI_PARAMETERS and
73  * - \ref ALIB_CLI_EXIT_CODES,
74  *
75  * When bootstrapping \alibmod_nolink_cli, method \alib{cli,CLIApp::DefineExitCodes} has to be
76  * invoked for (each) enum type.
77  *
78  * @tparam TEMD The type of the enum meta data tuple.
79  */
80 template<typename TEMD>
81 struct ArgumentDecl
82 {
83  /// The meta data tuple type.
84  using TTuple = typename lang::T_EnumMetaDataSpecification<TEMD>::TTuple;
85 
86  Enum EnumBox; ///< The enum element boxed in class \b %Enum.
87  TTuple& Tuple; ///< The resource tuple of the enum element.
88  Library& ResLib; ///< The library associated with the type \p{TEnum}.
89  ///< Used to load dependent resources.
90 
91  /**
92  * Templated constructor which takes an enum element of a custom enum type (enabled as
93  * described in class documentation).
94  *
95  * Stores the enum code, the tuple and the resource library.
96  *
97  * @param element The enum element
98  * @tparam TEnum C++ enum type enabled with \ref ALIB_CLI_EXIT_CODES.
99  */
100  template<typename TEnum>
101  inline
102  ArgumentDecl( TEnum element )
103  : EnumBox( element )
104  , Tuple ( *EnumMetaData<TEnum>::GetSingleton()->Get( element ) )
105  , ResLib ( lang::T_Resourced<TEnum>::Lib() )
106  {}

465 /**
466  * An option of a cli application. Options are read "automatically" using their declaration
467  * information defined with the resources accessed through the enum meta data of enum elements
468  * of custom types.
469  *
470  * However, such automatic read is limited due to the fact, that the simple values and flags
471  * defined with \alib{cli,OptionDecl}, can not provide the flexibility needed to perfectly
472  * parse options with a complex syntax.
473  *
474  * In this case, the way out is to use custom code that invokes \alib{cli,CLIApp,ReadOptions}
475  * and then processes all options that may have remaining arguments left in the list.
476  * Using field #Position further arguments may be consumed from \alib{cli,CLIApp::ArgsLeft}.<br>
477  * Note, "processing all options" may mean a nested loop. The outer is over the option types
478  * of \alib{cli,CLIApp,OptionsFound}, the inner is over the vector of options per type.
479  *
480  */
481 struct Option : public ArgumentDef
482 {
483  /// Arguments belonging to this option.
484  std::vector<String> Args;
485 
486  /// The declaration struct.
487  OptionDecl* Declaration = nullptr;
488 
489  /// Expose parent classes' constructor.
490  using ArgumentDef::ArgumentDef;
491 
492  /**
493  * Tries to reads the object from the current CLI arg(s).
494  * \note
495  * Unlike the read methods \alib{cli,Command::Read} and \alib{cli,Parameter::Read}, this
496  * method expects the argument to test not only by number with \p{argNo} but as well
497  * with string parameter \p{arg}.<br>
498  * This redundancy is needed is needed to easily implement shortcut options, that just
499  * replace a shortcut option read to another one, probably with a preset argument included.
500  *
501  * @param decl The declaration used for reading.
502  * @param arg The argument string starting with one or two hyphens.
503  * @param argNo The position of reading.
504  * @return The \c true on success, \c false otherwise.
505  */
506  ALIB_API
507  bool Read( OptionDecl& decl, String& arg, const size_t argNo );
