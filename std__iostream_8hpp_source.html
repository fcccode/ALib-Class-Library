41  /** ****************************************************************************************
42  * Parameter class used for \e application to objects of type
43  * \alib{strings,AStringBase,AString} with specialization of template struct
44  * <b>T_Apply<ISReadLine,TChar></b>.<br>
45  * Reads a line of text from a \b std::istream and appends it to the target \b %AString.<br>
46  * While, of-course, this class can be created 'inline' (similar objects of parameter classes
47  * found in \alib{strings,FormatBase,Format}), if a series of lines are to be read
48  * from a \b std::istream, a local object of this type might be created outside the loop
49  * and used inside.
50  * In this case, the 'output parameter' #IsEOF, can be used e.g. as a loop criteria.
51  * \see
52  * - \ref aworx::lib::strings::thirdparty::std::operator>>(std::istream&,NAString& string) "operator>>(istream&, AString&)" and
53  * \ref aworx::lib::strings::thirdparty::std::operator<<(std::ostream&,const NString&) "operator<<(ostream&, String&)".
54  * - For a sample, refer to source code of \alib class \b %IniFile, method
55  * \ref aworx::lib::config::IniFile::ReadFile "IniFile::ReadFile".
56  *
57  * @tparam TChar The character type of the input stream and string.
58  ******************************************************************************************/
59  template<typename TChar>
60  class ISReadLineBase
61  {
62  public:
63 
64  /// The input stream to read from.
65  ::std::basic_istream<TChar>* IStream;
66 
67  /// If \c CurrentData::KEEP, the target \c %AString is not cleared before the read operation.
68  lang::CurrentData TargetData;
69 
70  /// The amount of characters that the buffer is increased while reading parts of the line.
71  integer BufferSize;
72 
73  /// The maximum length of a single line to be read. Longer lines get truncated.
74  integer MaxLineWidth;
75 
76  /**
77  * Indicates if the end of the input stream was detected with the last read operation.
78  * If so, a next read operation will not change the string (or clear it, if #TargetData is
79  * \c false
80  */
