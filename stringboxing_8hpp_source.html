263 /**
264  * This interface class exposes interface method \c Invoke which writes the content of the
265  * box to the given \b %AString object.
266  * This base interface class, implements a default behaviour for all boxable types which are not
267  * equipped with a specific implementation of this class and hence, a singleton hereof is
268  * attached to the <b>%%ALib %Boxing</b> using
269  * \ref aworx::lib::boxing::Boxer::DefineDefaultInterface "Boxer::DefineDefaultInterface"
270  *
271  * Variant
272  * \ref aworx::lib::strings::boxing::IApply_TApplicable "IApply_TApplicable"
273  * can be attached without changes to any Boxer of custom types that implement the TMP interface
274  * introduced by \alib{strings,AStringBase::Apply,AString::Apply}.
275  */
276 template<typename TChar>
277 class IApply : public Interface
278 {
279  public:
280  /**
281  * Declares a static singleton used for the default interface registration and returns that.
282  * \note We do not inherit template class Singleton here, because each of our descendants
283  * needs to do this as well, which would lead to some confusion.
284  * @return A singleton instance of myself.
285  */
286  static IApply* GetDefaultSingleton()
287  {
288  static IApply me;
289  return &me;
290  }
291 
292  /// Constructor providing our runtime type to the parent.
293  IApply() : Interface( typeid(IApply<TChar>) ) {}
294 
295  /**
296  * The interface method. This default implementation writes out the raw value of the box
297  * in hexadecimal format. For standard types, such raw value reflect the memory address of
298  * the object boxed.<br>
299  * In debug-compilations, in addition the type name of the boxed value is written.
300  * @param box The box that the interface was invoked on.
301  * @param target The target \b %AString.
302  */
303  ALIB_API
304  virtual void Invoke( const Box& box, AStringBase<TChar>& target );
305 };

313 /**
314  * Templated implementation of #IApply for boxable types which are \e 'applicable'
315  * to AStrings. Applicable are all types that have a corresponding specialization of template struct
316  * \ref aworx::lib::strings::T_Apply. Consult the \alib String module documentation for more
317  * information.
318  *
319  * Once a type is made applicable, then on the boxing side of things, this interface class
320  * with template parameter \p{TApplicable} set to the type in question, may be attached to the
321  * boxed type.
322  */
323 template<typename TChar, typename TApplicable>
324 class IApply_TApplicable : public IApply<TChar>, public Singleton<IApply_TApplicable<TChar, TApplicable>>
325 {
326  public:
327  /**
328  * The interface method, writing the applicable type.
329  * @param box The box that the interface was invoked on.
330  * @param target The target \b %AString.
331  */
332  virtual inline void Invoke( const Box& box, AStringBase<TChar>& target ) override
333  {
334  target.template _<false>( box.Unbox<TApplicable>() );
335  }
336 };
