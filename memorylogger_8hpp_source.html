27 /** ************************************************************************************************
28  * A logger that logs all messages to an in-memory buffer of type AString. The name of the \e Logger
29  * defaults to "MEMORY".
30  **************************************************************************************************/
31 class MemoryLogger : public aworx::lib::lox::core::textlogger::PlainTextLogger
32 {
33  // #############################################################################################
34  // public fields
35  // #############################################################################################
36  public:
37  /**
38  * The logging Buffer. This can be accessed publicly and hence used as preferred.
39  * Especially, the whole log can easily be cleared using
40  * \alib{strings,AStringBase::Clear,AString::Clear}.
41  * In multi-threaded environments, \c Lox interfaces' mutex should be acquired
42  * before accessing this buffer. The initial size of the buffer is 8kb.
43  */
44  AString MemoryLog;
45 
46  /**
47  * If this field is set to \c true (which is the default), the effective length of the
48  * messages when converted to wide character strings are taken into account.
49  *
50  * Switching this off increases the overall logging performance (especially when logging
51  * into memory) significantly.
52  */
53  bool UseWStringLengthForTabAdjustments = true;

64  * Defaults to \c true.

92  * @param phase Indicates the beginning or end of a log line.
93  * @return Always returns true.
94  ******************************************************************************************/
95  virtual bool notifyLogOp(lib::lang::Phase phase)
96  {
97  // append new line if buffer has already lines stored
98  if ( phase == lib::lang::Phase::Begin && MemoryLog.IsNotEmpty() )
99  MemoryLog.NewLine();
100  return true;

103  /** ****************************************************************************************
104  * Write the given region of the given AString to the destination buffer.
105  *
106  * @param buffer The string to write a portion of.
107  * @param start The start of the portion in \p{buffer} to write out.
108  * @param length The length of the portion in \p{buffer} to write out.
109  * @return The number of characters written, -1 on error.
110  ******************************************************************************************/
111  virtual integer logSubstring( const String& buffer, integer start, integer length )
112  {
113  MemoryLog._<false>( buffer, start, length );
114  return UseWStringLengthForTabAdjustments
115  ? buffer.Substring<false>( start, length ).WStringLength()
116  : length;
117  }
