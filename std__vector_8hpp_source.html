173  /** ********************************************************************************************
174  * Copies or appends a region of a given \alib{strings,StringBase,String} into
175  * a given object of type \c std::vector<char>.
176  * The region is adjusted to the length of the %String.
177  *
178  * @param src The \alib{strings,StringBase,String} to convert into a std::vector<char>.
179  * @param target A result \c std::vector<char> to copy the specified region into.
180  * @param regionStart The start index of the region to be copied.
181  * @param regionLength The maximum length of the region to be copied.
182  * Defaults to \b MaxLen.
183  * @param targetData If \c CurrentData::Keep, the target \c std::vector<char> is not cleared before
184  * the result is written. Defaults to \c CurrentData::Clear.
185  *
186  * @return The (modified) \c std::vector<char>& result that was provided (for concatenation of calls).
187  **********************************************************************************************/
188  inline
189  std::vector<char>& ToStdVector( const NString& src,
190  std::vector<char>& target,
191  integer regionStart,
192  integer regionLength = strings::MaxLen,
193  lang::CurrentData targetData = lang::CurrentData::Clear )
194  {
195 
196  if ( targetData == lang::CurrentData::Clear )
197  target.clear();
198 
199  // adjust range, if empty return empty vector<char>
200  if ( src.AdjustRegion( regionStart, regionLength ) )
201  return target;
202 
203  // copy our buffer into target
204  target.reserve( static_cast<size_t>( regionLength ) );
205  const char* ps= src.Buffer() + regionStart;
206  const char* pe= ps + regionLength;
207  while(ps < pe )
208  target.push_back( *ps++ );
209  return target;
210  }
