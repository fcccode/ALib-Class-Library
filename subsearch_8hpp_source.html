27 /** ************************************************************************************************
28  * Implements "Knuth-Morris-Pratt" algorithm for searching a substring within a string.
29  *
30  * While the well known "Boyer-Moore-Algorithm" is even faster in the average case, for
31  * uni-code characters its implementation would be efficient only with very long haystack strings.
32  *
33  * For convenience, the following alias type names are available:
34  * - ref aworx::SubstringSearch,
35  * - ref aworx::NSubstringSearch and
36  * - ref aworx::WSubstringSearch.
37  *
38  * @tparam TChar The character type of the haystack and needle strings.
39  * @tparam TSensitivity The letter case sensitivity of the search..
40  **************************************************************************************************/
41 template<typename TChar, lang::Case TSensitivity= lang::Case::Sensitive>
42 class SubstringSearchBase
43 {
44  protected:
45  /// The needle set Knuth-Morris-Pratt prefix length table.

55  /** ****************************************************************************************
56  * Constructor. Passes the optional parameters to method #Compile.
