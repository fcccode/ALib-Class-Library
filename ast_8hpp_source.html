18  * Base class for nodes of abstract syntax trees of module \alibmod_expressions.
19  */
20 struct AST
21 {
22  /// The type of node.
23  enum class Types
24  {
25  Literal ,
26  Identifier ,
27  Function ,
28  UnaryOp ,
29  BinaryOp ,
30  TernaryOp ,
31  };
32 
33  /// Type of derived this AST node
34  Types NodeType;
35 
36  /// Position in original expression string.
37  integer Position;
38 
39  /// Deleted default constructor.
40  AST() = delete;
41 
42  /**
43  * Constructor
44  * @param type The node type.
45  * @param position The index of this AST in the expression string.
46  */
47  AST( Types type, integer position )
48  : NodeType(type)
49  , Position(position)
50  {}
51 
52  /// Virtual destructor.
53  virtual ~AST()
54  {}
55 
56  /**
57  * Recursively compiles nested nodes and invokes one of the add-methods of program for itself.
58  * @param program The program to be compiled.
59  * @param[out] normalized The normalized string, built during recursive compilation of the AST.
60  */
61  virtual void Assemble( Program& program, AString & normalized ) = 0;
62 
63  /**
64  * Recursively walks through the tree and performs optimizations, dependent on given flags.
65  *
66  * As of today, the only optimization performed in this AST itself is to combine nested unary
67  * '+' and '-' operators on literals.
68  * @param normalization The compiler flags denoting the normalization settings.
69  * @return A potentially replaced AST or itself.
70  */
71  virtual AST* Optimize( Normalization normalization ) = 0;
72 };

76  * Abstract syntax tree node representing identifiers.
77  */
78 struct ASTLiteral : public AST
79 {
80  /**
81  * This is a list of flags that store information obbout the format of the literal given
82  * in the expression string. This information is used with the generation of a normalized
83  * version of the literal.
84  */
85  enum class NFHint
86  {
87  NONE , ///< No hint.
88  Scientific , ///< Float was given in scientific format
89  Hexadecimal , ///< Integral value was given in hexadecimal format.
90  Octal , ///< Integral value was given in octal format.
91  Binary , ///< Integral value was given in binary format.
92  };
93 
94  Box Value; ///< The value of the literal.
95  NFHint Format; ///< The value of the literal.
96  AString stringValue; ///< Used with string literals as string copy buffer.
97 
98  /**
99  * Constructs a string literal.
100  * @param value The value of the literal. Will be copied to #stringValue, which then is
101  * itself placed in #Value.
102  * @param position The index of this AST in the expression string.
103  */
104  ASTLiteral( const String& value, integer position )
105  : AST(Types::Literal, position)
106  , Format(NFHint::NONE)
107  {
108  stringValue << value;
109  Value= stringValue;
110  }
111 
112  /**
113  * Constructs a string literal from a movable \b %AString.
114  * @param value The value of the literal. Will be moved to #stringValue, which then is
115  * itself placed in #Value.
116  * @param position The index of this AST in the expression string.
117  */
118  ASTLiteral( AString&& value, integer position )
119  : AST(Types::Literal, position)
120  , Format(NFHint::NONE)
121  {
122  stringValue= value;
123  Value= stringValue;
124  }
125 
126  /**
127  * Constructs an integer literal.
128  * @param value The value of the literal.
129  * @param position The index of this AST in the expression string.
130  * @param hint A hint about the format that expression string used to express the literal.
131  */
132  ASTLiteral( integer value, integer position, NFHint hint= NFHint::NONE )
133  : AST(Types::Literal, position)
134  , Format(hint)
135  {
136  Value= value;
137  }
138 
139  /**
140  * Constructs a floating point literal.
141  * @param value The value of the literal.
142  * @param position The index of this AST in the expression string.
143  * @param hint Optional information about the number format that was detected while parsing.
144  */
145  ASTLiteral( double value, integer position, NFHint hint= NFHint::NONE )
146  : AST(Types::Literal, position)
147  , Format(hint)
148  {
149  Value= value;
150  }
151 
152  /**
153  * Implements abstract method.
154  * @param program The program to be compiled.
155  * @param[out] normalized The normalized string, built during recursive compilation of the AST.
156  */
157  virtual void Assemble( Program& program, AString & normalized ) override;
158 
159  /**
160  * Implements abstract method.
161  * @param normalization The compiler flags denoting the normalization settings.
162  * @return A potentially replaced AST or itself.
163  */
164  virtual AST* Optimize( Normalization normalization ) override;
165 };
