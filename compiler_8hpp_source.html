155  /**
156  * Returns the precedence of given operator \b symbol.
157  *
158  * @param symbol The operator symbol.
159  * @return The precedence of the operator as defined with #AddBinaryOperator.
160  */
161  int GetBinaryOperatorPrecedence( const String& symbol )
162  {
163  auto it= BinaryOperators.find( symbol );
164  if( it == BinaryOperators.end() )
165  {
166  auto aliasOp= AlphabeticBinaryOperatorAliases.find( symbol );
167 
168  ALIB_ASSERT_ERROR( aliasOp != AlphabeticBinaryOperatorAliases.end(),
169  "Unknown binary operator '{}'.", symbol )
170 
171  it= BinaryOperators.find( aliasOp->second );
172  ALIB_ASSERT_ERROR( it != BinaryOperators.end(),
173  "Unknown binary operator '{}' which was aliased by '{}'.",
174  aliasOp->second, symbol )
175  }
176  return it->second;
177  }

418  /** ****************************************************************************************
419  * Firstly, this method parses the given expression string thereby builds an intermediate
420  * internal \alib{expressions,detail::AST,abstract syntax tree}.
421  * This tree is then translated into a \alib{expressions,detail::Program} which evaluates
422  * the expression in respect to a \alib{expressions,Scope} when executed.
423  *
424  * For building the program, attached \alib{expressions,CompilerPlugin} instances
425  * are queried.
426  *
427  * During this process, a \alib{expressions,Normalization,normalized} version of the input
428  * string is always created. While the original input string can
429  * be accessed with \alib{expressions,Expression::GetOriginalString}, the normalized version
430  * can be retrieved after the compilation with method
431  * \alib{expressions,Expression::GetNormalizedString}. During compilation, the normalization
432  * rules (flags) stored in field #CfgNormalization are applied.
433  *
434  * For compilation, a compile-time scope object is created by calling virtual method
435  * #getCompileTimeScope. If not overwritten, a standard scope object is used.
436  * If custom callbacks exists that are invokable at compile time
437  * (to perform program optimization) and that at the same time rely on custom mechanics
438  * provided with a custom scope class, then a custom, derived version of this class
439  * has to be used that overrides method #getCompileTimeScope accordingly.
440  *
441  * @param expressionString The string to parse.
442  * @return A shared pointer to the compiled expression. Contains \c nullptr on failure.
443  ******************************************************************************************/

445  SPExpression Compile( const String& expressionString );
