34 /** ************************************************************************************************
35  * As explained in detail in the documentation of namespace \ref #aworx::lib::time, \alib
36  * supports a steady time model with class \alib{time,Ticks} and one representing the
37  * system clock with class \alib{time,DateTime}. Only values of the latter type can be converted
38  * to human readable (calendar) date and time values.
39  *
40  * In some situations however, a software that requires steady, monotonic time points, may also be
41  * required to present these time points in human readable format. It is of-course possible
42  * to do some sort of conversion. For that, simply both clocks need to be probed at the same point
43  * in time and then time points of both notions can be put in relation to these two probes.
44  *
45  * The effect however is that the conversion results will change for all values as soon as
46  * the system clock is changed and the probe values of the two clocks are updated.
47  * This is true also for values that are "older" than the point in time that the clock change
48  * happened.The reason is quickly understood: The system clock counter changes, while the steady
49  * counter does not.
50  *
51  * To give the user of \alib full control about how system clock changes are reflected,
52  * the conversion of time points is encapsulated by this class together with one pair of clock probe
53  * data. A software can use one ore more instances of this class and update (synchronize) these
54  * instances independently.
55  **************************************************************************************************/
56 class TicksConverter
57 {
58  protected:
59  /// Time point of steady clock of last invocation of #SyncClocks.
60  Ticks::TTimePoint steadyClockSyncTime;
61 
62  /// Time point of system clock of last invocation of #SyncClocks.
63  DateTime::TTimePoint systemClockSyncTime;
64 
65  public:
66  /** ****************************************************************************************
67  * Constructor. Invokes #SyncClocks.
68  ******************************************************************************************/
69  inline ALIB_CPP14_CONSTEXPR
70  TicksConverter()
71  {
72  SyncClocks();
73  }
74 
75  /** ****************************************************************************************
76  * Generates a set of "probes" of the steady, monotonic clock and the system clock.
77  * The measurement of both clocks is repeated the given number of times and the pair with
78  * the smallest difference between both is chosen.
79  * This approach mitigates the risk of using a pair for which thread execution had been
80  * interrupted during the two measurements.
81  *
82  * Note that after a call to this method, the conversion methods may return slightly
83  * different values than before the call, even if the system clock was not changed.
84  *
85  * If this method is not invoked after a change of the system clock, such change of the
86  * system clock is not reflected by the conversion methods.
87  * In other words, the conversion methods always work just as if the system clock had not
88  * changed since the last invocation of this method.
89  *
90  * \note
91  * On a GNU/Linux workstation (without workload), the error observed when doing only one
92  * measurement was in the magnitude of several microseconds.
93  *
94  * @param qtyRepeats The number of measurements to perform. Defaults to \c 5.
95  ******************************************************************************************/
96  ALIB_API
97  void SyncClocks( int qtyRepeats = 5 );
98 
99  /** ****************************************************************************************
100  * Sets the pair of conversion times to equal the other converter object. This is useful
101  * to avoid differences in conversion across converter instances used in a software.
102  *
103  * @param other Another converter object to copy the the synchronization information from.
104  ******************************************************************************************/
105  inline
106  void SetAs( const TicksConverter& other )
107  {
108  steadyClockSyncTime= other.steadyClockSyncTime;
109  systemClockSyncTime= other.systemClockSyncTime;
110  }
111 
112  /** ****************************************************************************************
113  * Converts a \b %Ticks object to a \b %DateTime object.
114  * @param ticks The ticks object to convert.
115  * @return The date time object.
116  ******************************************************************************************/
117  inline
118  DateTime ToDateTime( Ticks ticks )
119  {
120  DateTime result( Initialization::Suppress );
121  result.NativeValue()= systemClockSyncTime + std::chrono::duration_cast<std::chrono::system_clock::duration>( ticks.NativeValue() - steadyClockSyncTime );
122  return result;
123  }
124 
125  /** ****************************************************************************************
126  * Converts a \b %DateTime object to a \b %Ticks object.
127  * @param dateTime The date/time object to convert.
128  * @return The date time object.
129  ******************************************************************************************/
130  inline
131  Ticks ToTicks( DateTime dateTime )
132  {
133  Ticks result( Initialization::Suppress );
134  result.NativeValue()= (dateTime.NativeValue() - systemClockSyncTime ) + steadyClockSyncTime;
135  return result;
136 
137  }
138 };
