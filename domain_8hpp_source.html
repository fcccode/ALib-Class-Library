141  /** ****************************************************************************************
142  * Adds a new entry in field #Data and recursively demands the same from its sub-domains.
143  * Checks if a logger with the same name exists.

149  int AddLogger( core::Logger* logger)
150  {
151  // let our root do this
152  if ( Parent != nullptr )
153  return Parent->AddLogger( logger );

155  // check for doubles

250  /** ****************************************************************************************
251  * Sets the verbosity for a logger of this domain of all its sub domains to the specified

262  {
263  LoggerData& ld= Data[static_cast<size_t>(loggerNo)];
264  if( priority >= ld.Priority )

309  /** ****************************************************************************************
310  * Determines if the domain is active in respect to the given Verbosity.
311  *
312  * @param loggerNo The number of the \e Logger whose \e Verbosity is to be evaluated against
313  * \p{statement}.
314  * @param statement The \e Verbosity to check.
315  * @return \c true if domain is active (log should be performed)
316  ******************************************************************************************/
317  inline
318  bool IsActive( int loggerNo, Verbosity statement )
319  {
320  Verbosity domain= GetVerbosity( loggerNo );
321 
322  // domain ^ / stmnt > | Off Error Warning Info Verbose
323  // ---------------------------------------------------------------------
324  // Off | - - - - -
325  // Errors | - Y - - -
326  // Warning | - Y Y - -
327  // Info | - Y Y Y -
328  // Verbose | - Y Y Y Y
329 
330  if( statement != Verbosity::Off
331  && ( ( domain == Verbosity::Error && statement == Verbosity::Error )
332  || ( domain == Verbosity::Warning && ( statement == Verbosity::Warning || statement == Verbosity::Error ) )
333  || ( domain == Verbosity::Info && statement != Verbosity::Verbose )
334  || domain == Verbosity::Verbose )
335  )
336  {
337  Data[static_cast<size_t>(loggerNo)].LogCallsPerDomain++;
338  return true;
339  }
340 
341  return false;
342  }
