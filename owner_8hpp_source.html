41 /** ************************************************************************************************
42  * Ensures that an object of template type \p{TOwnable} is acquired and properly released when
43  * unwinding the stack. This class is meant to be allocated only on the stack and not on the heap.
44  * Therefore, the new operators are declared private.
45  *
46  * In debug compilations the constructor expects caller source information parameters
47  * \p{file}, \p{line} and \p{func}. It is convenient to use macro
48  * \ref ALIB_CALLER_PRUNED to provide those.
49  *
50  * \see
51  * Preprocessor macros \ref ALIB_OWN, \ref ALIB_LOCK and \ref ALIB_LOCK_WITH for a convenient way to
52  * use this class.
53  *
54  * @tparam TOwnable The type to own. The type needs to have methods \b %Acquire and \b %Release
55  * available.
56  **************************************************************************************************/
57 template <typename TOwnable>
58 struct Owner
59 {
60  // #############################################################################################
61  // Disallow new operations.
62  // #############################################################################################
63  private:
64  void* operator new (size_t); ///< Private declared new, to disallow heap allocation.
65  void* operator new (size_t, void*); ///< Private declared new, to disallow heap allocation.
66  void* operator new[](size_t); ///< Private declared new, to disallow heap allocation.
67  void* operator new[](size_t, void*); ///< Private declared new, to disallow heap allocation.
68  void operator = (const Owner& ); ///< Private declared assignment operator.
69 
70  protected:
71  /// All we own is this.
72  TOwnable& theOwnable;
73 
74  public:
75 
76  /** **************************************************************************************
77  * The constructor. Invokes Acquire() on the owner.
78  * @param ownable The ownable to acquire.
79  *
80  * @param file Caller information. Available only in debug compilations.
81  * @param line Caller information. Available only in debug compilations.
82  * @param func Caller information. Available only in debug compilations.
83  *****************************************************************************************/
84  #if ALIB_DEBUG
85  Owner( TOwnable& ownable, const NTString& file, int line, const NTString& func)
86  : theOwnable(ownable)
87  {
88  ownable.Acquire( file, line, func );
89  }
90  #else
91  Owner( TOwnable& ownable ) : theOwnable(ownable)
92  {
93  ownable.Acquire();
94  }
95  #endif
96 
97  /** **************************************************************************************
98  * The destructor. Releases the owner by invoking Release().
99  *****************************************************************************************/
100  ~Owner() { theOwnable.Release(); }
